api.js接口文件

  1. 封装api.js接口文件

    // 引入axios
import axios from 'axios'

// 路由地址前缀
// 如：http://192.168.7.74:8080/prod-api/unify/api/queryQSGZJWSJDSTSDH
const APIPRE = '/prod-api'

// 添加请求拦截器
axios.interceptors.request.use(function (config) {
  // 双向http交互协议时，客户端接收数据乱码，使用以下方式解决，在请求头中设置客户端要获取的数据类型
  config.headers['Content-Type'] = 'application/json;charset=UTF-8'
  // 请求时间不能超过50000毫秒
  config['timeout'] = 50000
  // 请求拦截后，必须对config进行返回
  return config
}, function (error) {
  // Do something with request error
  return Promise.reject(error)
})

// 添加响应拦截器
// axios.interceptors.response.use(function(response) {
//   return response
// }, function(error) {
//   if (error.response) {
//     Vue.prototype.$toast('请求出错')
//   } else {
//     Vue.prototype.$toast('请求超时')
//   }
//   return Promise.reject(error)
// })

 // 导出get请求和post请求
export default {
  /**
   * get请求
   * @param url 请求路径
   * @param param 入参
   * @param headers 请求头
   * @returns {Promise<any>}
   */
   // 导出get请求
  get (url, param = {}, headers = {}) {
    console.info('请求参数：', param)
    return new Promise((resolve, reject) => {
      // 使用get请求，APIPRE + url将路由地址前缀APIPRE与请求路径url进行拼接
      return axios.get(APIPRE + url, param, {
        headers: headers
      }).then(res => {
        console.info('请求返回：', res.data)
        // 请求成功
        if (res.data.code === 200) {
          resolve(res.data)
        } else {
          reject(res.data)
        }
      }).catch(err => {
        // 请求失败
        if (err.response && err.response.status != '504') {
          reject({ message: '请求接口报错！错误码：' + err.response.status })
        } else {
          reject({ message: '请求超时，请重试。' })
        }
      })
    })
  },
  /**
   * post 请求
   * @param url 请求路径
   * @param data 入参
   * @param headers 请求头
   * @returns {Promise<any>}
   */
  // 导出post请求
  post (url, data = {}, headers = {}) {
    console.info('请求参数：', data)
    return new Promise((resolve, reject) => {
      // 使用post请求，APIPRE + url将路由地址前缀APIPRE与请求路径url进行拼接
      return axios.post(APIPRE + url, data, {
        headers: headers
      }).then(res => {
        console.info('请求返回：', res.data)
        // 请求成功
        if (res.data.code === 200) {
          resolve(res.data)
        } else {
          reject(res.data)
        }
      }).catch(err => {
        // 请求失败
        if (err.response && err.response.status !== '504') {
          reject({ msg: '请求接口报错！错误码：' + err.response.status })
        } else {
          reject({ msg: '请求超时，请重试。' })
        }
      })
    })
  }
}

  2. 使用api.js接口文件：在main.js入口文件中写以下代码

    // 引入api.js文件
import api from './***/api'

Vue.prototype.$http = api

  3. 之后在组件中需要通过接口发送数据请求时，直接使用this.$http.post或者this.$http.get即可，例

    // 发送post请求
this.$http.post('/unify/api/queryWFWGZYYLJGMD', {
    // 发送请求数据
    DGS_YLJGMC: this.keyword
  }).then(res => {
    ...
  }, err => {
    ...
  })
}
